#### **Problema a Resolver:**
La industria del Gas Natural Licuado (LNG) enfrenta dos desafíos críticos:
1. **Pérdidas por Boil-Off Gas (BOG):** Durante el transporte marítimo, el LNG se evapora naturalmente (0.1-0.3%/día), representando pérdidas económicas de millones de dólares anuales y emisiones de metano significativas .
2. **Ineficiencia en Rutas Marítimas:** Las rutas subóptimas incrementan el consumo de combustible, tiempo de tránsito y exposición a condiciones oceanográficas adversas .

#### **Justificación del Impacto:**
- **Económico:** Reducción del 15-25% en pérdidas por BOG mediante predicción precisa .
- **Ambiental:** Disminución de emisiones de CO₂ y metano, alineado con regulaciones IMO 2025 .
- **Operacional:** Optimización de rutas en tiempo real considerando corrientes, oleaje y condiciones meteorológicas .
- **Seguridad:** Evaluación de riesgos marítimos mediante Redes Bayesianas para prevención de accidentes .

"""

Módulo para ingesta de datos desde fuentes oficiales (NOAA, Copernicus, IMO)

class NOAADataIngestor:
    """Ingestor de datos oceanográficos de NOAA"""
    
    def __init__(self, api_key: Optional[str] = None):
        self.api_key = api_key or os.getenv('NOAA_API_KEY', '')
        self.base_url = CONFIG.data.NOAA_API_BASE
        self.headers = {'token': self.api_key} if self.api_key else {}
        
    def fetch_buoy_data(self, station_id: str, start_date: str, 
                       end_date: str) -> pd.DataFrame:
        """
        Obtiene datos de boyas NDBC
        
        Args:
            station_id: ID de la boya (ej. '41002')
            start_date: Fecha inicio (YYYY-MM-DD)
            end_date: Fecha fin (YYYY-MM-DD)
        """
        url = f"https://www.ndbc.noaa.gov/data/realtime2/{station_id}.spec"
        
        try:
            response = requests.get(url, timeout=30)
            response.raise_for_status()
            
            # Parsear datos NDBC
            lines = response.text.strip().split('\n')
            data = []
            
            for line in lines[1:]:  # Saltar header
                parts = line.split()
                if len(parts) >= 15:
                    data.append({
                        'timestamp': f"{parts[0]}-{parts[1]}-{parts[2]} {parts[3]}:{parts[4]}",
                        'wave_height': float(parts[5]) if parts[5] != 'MM' else np.nan,
                        'wave_period': float(parts[6]) if parts[6] != 'MM' else np.nan,
                        'wind_speed': float(parts[10]) if parts[10] != 'MM' else np.nan,
                        'wind_direction': float(parts[11]) if parts[11] != 'MM' else np.nan,
                        'air_temperature': float(parts[13]) if parts[13] != 'MM' else np.nan,
                        'water_temperature': float(parts[14]) if parts[14] != 'MM' else np.nan
                    })
            
            df = pd.DataFrame(data)
            df['timestamp'] = pd.to_datetime(df['timestamp'])
            return df
            
        except Exception as e:
            print(f"Error fetching buoy  {e}")
            return pd.DataFrame()
    
    def fetch_ocean_temperature(self, lat: float, lon: float, 
                                depth: float = 0) -> pd.DataFrame:
        """Obtiene temperatura oceánica para coordenadas específicas"""
        # Usar ERDDAP de NOAA
        url = "https://coastwatch.pfeg.noaa.gov/erddap/tabledap/cwssstday.csv"
        params = {
            'time': '2020-01-01T00:00:00Z,2025-12-31T23:59:59Z',
            'latitude': f'{lat-1},{lat+1}',
            'longitude': f'{lon-1},{lon+1}',
            'variables': 'sst'
        }
        
        try:
            response = requests.get(url, params=params, timeout=60)
            response.raise_for_status()
            df = pd.read_csv(pd.io.common.StringIO(response.text))
            return df
        except Exception as e:
            print(f"Error fetching ocean temperature: {e}")
            return pd.DataFrame()

class AISDataIngestor:
    """Ingestor de datos AIS (Automatic Identification System)"""
    
    def __init__(self, data_path: str = None):
        self.data_path = data_path or CONFIG.data.AIS_DATA_PATH
        
    def load_ais_trajectory(self, mmsi: int, 
                           start_date: str, 
                           end_date: str) -> pd.DataFrame:
        """
        Carga trayectorias AIS para un buque específico
        
        Args:
            mmsi: Maritime Mobile Service Identity del buque
            start_date: Fecha inicio
            end_date: Fecha fin
        """
        # Buscar archivos AIS
        ais_files = list(Path(self.data_path).glob(f"*{mmsi}*.csv"))
        
        if not ais_files:
            print(f"No AIS files found for MMSI {mmsi}")
            return pd.DataFrame()
        
        dfs = []
        for file in ais_files:
            try:
                df = pd.read_csv(file)
                dfs.append(df)
            except Exception as e:
                print(f"Error reading {file}: {e}")
        
        if dfs:
            combined = pd.concat(dfs, ignore_index=True)
            combined['timestamp'] = pd.to_datetime(combined['timestamp'])
            mask = (combined['timestamp'] >= start_date) & \
                   (combined['timestamp'] <= end_date)
            return combined[mask]
        
        return pd.DataFrame()
    
    def extract_vessel_features(self, ais_df: pd.DataFrame) -> Dict:
        """Extrae características del buque desde datos AIS"""
        if ais_df.empty:
            return {}
        
        return {
            'mmsi': ais_df['mmsi'].iloc[0] if 'mmsi' in ais_df.columns else None,
            'total_distance_km': self._calculate_distance(ais_df),
            'avg_speed_knots': ais_df['sog'].mean() if 'sog' in ais_df.columns else None,
            'max_speed_knots': ais_df['sog'].max() if 'sog' in ais_df.columns else None,
            'course_changes': self._count_course_changes(ais_df),
            'ports_visited': len(ais_df['destination'].unique()) if 'destination' in ais_df.columns else 0
        }
    
    def _calculate_distance(self, df: pd.DataFrame) -> float:
        """Calcula distancia total recorrida (km)"""
        if len(df) < 2:
            return 0.0
        
        from math import radians, sin, cos, sqrt, atan2
        
        def haversine(lat1, lon1, lat2, lon2):
            R = 6371  # Radio terrestre km
            dlat = radians(lat2 - lat1)
            dlon = radians(lon2 - lon1)
            a = sin(dlat/2)**2 + cos(radians(lat1)) * cos(radians(lat2)) * sin(dlon/2)**2
            c = 2 * atan2(sqrt(a), sqrt(1-a))
            return R * c
        
        total = 0
        for i in range(1, len(df)):
            if 'latitude' in df.columns and 'longitude' in df.columns:
                total += haversine(
                    df.iloc[i-1]['latitude'], df.iloc[i-1]['longitude'],
                    df.iloc[i]['latitude'], df.iloc[i]['longitude']
                )
        return total
    
    def _count_course_changes(self, df: pd.DataFrame, threshold: float = 10) -> int:
        """Cuenta cambios significativos de rumbo"""
        if 'cog' not in df.columns or len(df) < 2:
            return 0
        
        changes = 0
        for i in range(1, len(df)):
            diff = abs(df.iloc[i]['cog'] - df.iloc[i-1]['cog'])
            if diff > 180:
                diff = 360 - diff
            if diff > threshold:
                changes += 1
        return changes


class LNGTerminalDataIngestor:
    """Ingestor de datos de terminales LNG"""
    
    def __init__(self, data_path: str = None):
        self.data_path = data_path or CONFIG.data.LNG_TERMINAL_DATA
        
    def load_bog_data(self, terminal_id: str) -> pd.DataFrame:
        """Carga datos históricos de Boil-Off Gas"""
        file_path = Path(self.data_path) / f"{terminal_id}_bog.csv"
        
        if not file_path.exists():
            print(f"BOG data file not found: {file_path}")
            return pd.DataFrame()
        
        df = pd.read_csv(file_path)
        df['timestamp'] = pd.to_datetime(df['timestamp'])
        return df
    
    def generate_synthetic_bog_data(self, num_days: int = 365, 
                                    tank_volume: float = 170000) -> pd.DataFrame:
        """
        Genera datos sintéticos de BOG para testing
        
        Basado en modelos físicos de transferencia de calor 
        """
        dates = pd.date_range(start='2025-01-01', periods=num_days, freq='D')
        
        # Factores que afectan BOG
        ambient_temp = 15 + 10 * np.sin(np.linspace(0, 2*np.pi, num_days))  # Variación estacional
        sea_temp = 12 + 5 * np.sin(np.linspace(0, 2*np.pi, num_days))
        wind_speed = np.random.uniform(5, 25, num_days)
        
        # Modelo simplificado de BOG
        # BOG rate = base_rate + f(ambient_temp, sea_temp, wind, insulation)
        base_rate = CONFIG.lng.BOG_BASELINE_RATE
        temp_factor = 0.0001 * (ambient_temp - sea_temp)
        wind_factor = 0.00005 * wind_speed
        noise = np.random.normal(0, 0.0002, num_days)
        
        bog_rate = base_rate + temp_factor + wind_factor + noise
        bog_rate = np.clip(bog_rate, 0.0005, CONFIG.lng.BOG_MAX_ACCEPTABLE)
        
        bog_volume = bog_rate * tank_volume  # m³/día
        
        df = pd.DataFrame({
            'timestamp': dates,
            'ambient_temperature': ambient_temp,
            'sea_temperature': sea_temp,
            'wind_speed': wind_speed,
            'bog_rate': bog_rate,
            'bog_volume_m3': bog_volume,
            'tank_pressure_bar': 0.15 + 0.05 * np.random.uniform(0, 1, num_days)
        })
        
        return df


class MaritimeDataPipeline:
    """Pipeline principal de ingesta de datos"""
    
    def __init__(self, noaa_key: str = None, 
                 copernicus_user: str = None,
                 copernicus_pass: str = None):
        self.noaa = NOAADataIngestor(noaa_key)
        self.copernicus = CopernicusDataIngestor(copernicus_user, copernicus_pass) \
            if copernicus_user and copernicus_pass else None
        self.ais = AISDataIngestor()
        self.lng = LNGTerminalDataIngestor()
        
    def run_full_ingestion(self, region: Dict, 
                          date_range: Tuple[str, str],
                          vessel_mmsi: int = None) -> Dict[str, pd.DataFrame]:
        """
        Ejecuta ingesta completa de datos
        
        Args:
            region: {'lat_min', 'lat_max', 'lon_min', 'lon_max'}
            date_range: (start_date, end_date)
            vessel_mmsi: MMSI del buque de interés
        """
        results = {}
        
        # 1. Datos oceanográficos NOAA
        print("Fetching NOAA buoy data...")
        results['noaa_buoys'] = self.noaa.fetch_buoy_data(
            station_id='41002',  # Ejemplo: boya Atlántico
            start_date=date_range[0],
            end_date=date_range[1]
        )
        
        # 2. Datos AIS si se proporciona MMSI
        if vessel_mmsi:
            print(f"Fetching AIS data for MMSI {vessel_mmsi}...")
            results['ais_trajectory'] = self.ais.load_ais_trajectory(
                mmsi=vessel_mmsi,
                start_date=date_range[0],
                end_date=date_range[1]
            )
            results['ais_features'] = pd.DataFrame([
                self.ais.extract_vessel_features(results['ais_trajectory'])
            ])
        
        # 3. Datos BOG de terminal LNG
        print("Loading LNG terminal BOG data...")
        results['bog_data'] = self.lng.generate_synthetic_bog_data(

"""
MARITIME-AI Hybrid Model Architecture
Arquitectura híbrida: GNN + PINN + Bayesian Networks
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
from typing import Dict, List, Tuple, Optional
from torch_geometric.nn import GATConv, GINConv, global_mean_pool
from torch_geometric.data import Data, DataLoader
import xarray as xr

class MaritimeGNN(nn.Module):
    """
    Graph Neural Network para optimización de rutas marítimas
    
    Usa Graph Attention Networks (GAT) para modelar:
    - Nodos: Puertos, waypoints, condiciones oceanográficas
    - Aristas: Rutas posibles, corrientes, restricciones
    """
    
    def __init__(self, node_features: int, edge_features: int, 
                 hidden_dim: int = None, num_layers: int = None,
                 num_heads: int = None, dropout: float = None):
        super().__init__()
        
        # Usar configuración global si no se proporciona
        hidden_dim = hidden_dim or CONFIG.model.GNN_HIDDEN_DIM
        num_layers = num_layers or CONFIG.model.GNN_NUM_LAYERS
        num_heads = num_heads or CONFIG.model.GNN_HEADS
        dropout = dropout or CONFIG.model.GNN_DROPOUT
        
        self.node_encoder = nn.Linear(node_features, hidden_dim)
        self.edge_encoder = nn.Linear(edge_features, hidden_dim)
        
        # Capas GAT (Graph Attention)
        self.gat_layers = nn.ModuleList()
        for i in range(num_layers):
            in_dim = hidden_dim if i == 0 else hidden_dim * num_heads
            self.gat_layers.append(
                GATConv(
                    in_channels=in_dim,
                    out_channels=hidden_dim,
                    heads=num_heads if i < num_layers - 1 else 1,
                    dropout=dropout,
                    concat=i < num_layers - 1
                )
            )
        
        # Capas fully connected para predicción
        self.fc_layers = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(hidden_dim // 2, hidden_dim // 4),
            nn.ReLU(),
            nn.Linear(hidden_dim // 4, 1)  # Score de ruta óptima
        )
        
    def forward(self, x: torch.Tensor, edge_index: torch.Tensor,
                edge_attr: torch.Tensor, batch: torch.Tensor) -> torch.Tensor:
        """
        Forward pass del GNN
        
        Args:
            x: Features de nodos [num_nodes, node_features]
            edge_index: Conectividad del grafo [2, num_edges]
            edge_attr: Features de aristas [num_edges, edge_features]
            batch: Batch assignment para graph pooling
        """
        # Encoding inicial
        x = self.node_encoder(x)
        edge_attr = self.edge_encoder(edge_attr)
        
        # Pasar por capas GAT
        for gat_layer in self.gat_layers:
            x = gat_layer(x, edge_index, edge_attr=edge_attr)
            x = F.relu(x)
        
        # Global pooling
        x = global_mean_pool(x, batch)
        
        # Predicción final
        output = self.fc_layers(x)
        return output


class PhysicsInformedNN(nn.Module):
    """
    Physics-Informed Neural Network para predicción de Boil-Off Gas
    
    Incorpora la ecuación de transferencia de calor como regularización:
    ∂T/∂t = α∇²T + Q/(ρ·cp)
    
    Donde:
    - T: Temperatura
    - α: Difusividad térmica
    - Q: Fuente de calor
    - ρ: Densidad
    - cp: Calor específico
    """
    
    def __init__(self, input_dim: int, hidden_layers: List[int] = None,
                 activation: str = None):
        super().__init__()
        
        hidden_layers = hidden_layers or CONFIG.model.PINN_HIDDEN_LAYERS
        activation = activation or CONFIG.model.PINN_ACTIVATION
        
        # Construir arquitectura
        layers = []
        prev_dim = input_dim
        
        for hidden_dim in hidden_layers:
            layers.append(nn.Linear(prev_dim, hidden_dim))
            if activation == "tanh":
                layers.append(nn.Tanh())
            elif activation == "relu":
                layers.append(nn.ReLU())
            elif activation == "swish":
                layers.append(nn.SiLU())
            prev_dim = hidden_dim
        
        layers.append(nn.Linear(prev_dim, 1))  # Output: BOG rate
        
        self.network = nn.Sequential(*layers)
        
        # Parámetros físicos (aprendibles)
        self.thermal_diffusivity = nn.Parameter(torch.tensor(1.0e-7))
        self.heat_transfer_coef = nn.Parameter(torch.tensor(10.0))
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """Forward pass estándar"""
        return self.network(x)
    
    def compute_physics_loss(self, x: torch.Tensor, 
                            predicted: torch.Tensor) -> torch.Tensor:
        """
        Calcula el término de pérdida basado en física
        
        Implementa la ecuación de calor como regularización
        """
        x.requires_grad_(True)
        predicted = self.network(x)
        
        # Gradientes automáticos
        grad_outputs = torch.ones_like(predicted)
        gradients = torch.autograd.grad(
            outputs=predicted,
            inputs=x,
            grad_outputs=grad_outputs,
            create_graph=True,
            retain_graph=True
        )[0]
        
        # Laplaciano (simplificado para 1D)
        laplacian = torch.zeros_like(predicted)
        for i in range(x.shape[1]):
            grad_i = gradients[:, i:i+1]
            grad_i.requires_grad_(True)
            laplacian_i = torch.autograd.grad(
                outputs=grad_i,
                inputs=x,
                grad_outputs=torch.ones_like(grad_i),
                create_graph=True
            )[0][:, i:i+1]
            laplacian += laplacian_i
        
        # Ecuación de calor residual
        # ∂T/∂t - α∇²T = 0 (en estado estacionario)
        physics_residual = -self.thermal_diffusivity * laplacian
        
        return torch.mean(physics_residual ** 2)
    
    def total_loss(self, x: torch.Tensor, y_true: torch.Tensor,
                   physics_weight: float = None) -> Tuple[torch.Tensor, Dict]:
        """
        Pérdida total: MSE + Physics regularization
        
        Args:
            x: Inputs (condiciones ambientales, propiedades tanque)
            y_true: BOG rate observado
            physics_weight: Peso del término físico
        """
        physics_weight = physics_weight or CONFIG.model.PINN_WEIGHT_PHYSICS
        
        predicted = self.forward(x)
        
        # MSE loss
        mse_loss = F.mse_loss(predicted, y_true)
        
        # Physics loss
        physics_loss = self.compute_physics_loss(x, predicted)
        
        # Loss total
        total_loss = mse_loss + physics_weight * physics_loss
        
        metrics = {
            'mse': mse_loss.item(),
            'physics': physics_loss.item(),
            'total': total_loss.item()
        }
        
        return total_loss, metrics


class BayesianRiskNetwork:
    """
    Red Bayesiana para evaluación de riesgos marítimos
    
    Variables:
    - Condiciones meteorológicas (viento, oleaje, visibilidad)
    - Estado del buque (velocidad, carga, mantenimiento)
    - Factores humanos (experiencia tripulación, fatiga)
    - Riesgo de accidente (colisión, varada, incendio)
    """
    
    def __init__(self):
        try:
            from pgmpy.models import BayesianNetwork
            from pgmpy.estimators import HillClimbSearch, BicScore
            from pgmpy.estimators import MaximumLikelihoodEstimator, BayesianEstimator
            self.HillClimbSearch = HillClimbSearch
            self.BicScore = BicScore
            self.MLE = MaximumLikelihoodEstimator
            self.BayesEst = BayesianEstimator
            self.BayesianNetwork = BayesianNetwork
            self.available = True
        except ImportError:
            print("pgmpy not installed. Install with: pip install pgmpy")
            self.available = False
    
    def create_maritime_risk_model(self) -> 'BayesianNetwork':
        """Crea la estructura de la red bayesiana"""
        if not self.available:
            return None
        
        # Definir aristas (relaciones causales)
        edges = [
            ('wind_speed', 'wave_height'),
            ('wind_speed', 'vessel_stability'),
            ('wave_height', 'vessel_stability'),
            ('wave_height', 'navigation_difficulty'),
            ('visibility', 'navigation_difficulty'),
            ('vessel_stability', 'accident_risk'),
            ('navigation_difficulty', 'accident_risk'),
            ('crew_experience', 'accident_risk'),
            ('maintenance_status', 'vessel_stability'),
            ('maintenance_status', 'accident_risk'),
            ('cargo_load', 'vessel_stability'),
            ('sea_current', 'navigation_difficulty'),
            ('accident_risk', 'route_safety_score')
        ]
        
        model = self.BayesianNetwork(edges)
        return model
    
    def fit_from_data(self, data: pd.DataFrame) -> 'BayesianNetwork':
        """Ajusta la red bayesiana desde datos"""
        if not self.available:
            return None
        
        import pandas as pd
        
        model = self.create_maritime_risk_model()
        
        # Aprendizaje de estructura (opcional, si no se define manualmente)
        # hc = self.HillClimbSearch(data)
        # best_model = hc.estimate(scoring_method=self.BicScore(data))
        
        # Estimación de parámetros
        model.fit(data, estimator=self.BayesEst, prior_type="BDeu")
        
        return model
    
    def query_risk(self, model: 'BayesianNetwork', 
                   evidence: Dict) -> Dict:
        """
        Consulta la probabilidad de riesgo dado evidencia
        
        Args:
            model: Red bayesiana entrenada
            evidence: Dict con variables observadas
        """
        if not self.available or model is None:
            return {}
        
        from pgmpy.inference import VariableElimination
        
        inference = VariableElimination(model)
        
        # Consultar probabilidad de accidente
        result = inference.query(
            variables=['accident_risk'],
            evidence=evidence
        )
        
        return {
            'low_risk': float(result.values[0]),
            'medium_risk': float(result.values[1]),
            'high_risk': float(result.values[2])
        }


class MaritimeHybridModel:
    """
    Modelo híbrido principal que integra GNN + PINN + Bayesian
    
    Orquesta los tres componentes para:
    1. Optimizar rutas (GNN)
    2. Predecir BOG (PINN)
    3. Evaluar riesgos (Bayesian)
    """
    
    def __init__(self, node_features: int = 10, edge_features: int = 5,
                 pinn_input_dim: int = 8):
        self.gnn = MaritimeGNN(node_features, edge_features)
        self.pinn = PhysicsInformedNN(pinn_input_dim)
        self.bayesian = BayesianRiskNetwork()
        
        self.gnn_optimizer = None
        self.pinn_optimizer = None
        
    def compile(self, gn_lr: float = None, pinn_lr: float = None):
        """Configura optimizadores"""
        gn_lr = gn_lr or CONFIG.model.LEARNING_RATE
        pinn_lr = pinn_lr or CONFIG.model.LEARNING_RATE
        
        self.gnn_optimizer = torch.optim.Adam(
            self.gnn.parameters(), lr=gn_lr
        )
        self.pinn_optimizer = torch.optim.Adam(
            self.pinn.parameters(), lr=pinn_lr
        )
    
    def train_gnn(self, dataloader: DataLoader, num_epochs: int = None,
                  early_stop_patience: int = None) -> List[Dict]:
        """Entrena el componente GNN"""
        num_epochs = num_epochs or CONFIG.model.NUM_EPOCHS
        early_stop_patience = early_stop_patience or CONFIG.model.EARLY_STOPPING_PATIENCE
        
        self.gnn.train()
        history = []
        best_loss = float('inf')
        patience_counter = 0
        
        for epoch in range(num_epochs):
            epoch_loss = 0
            num_batches = 0
            
            for batch in dataloader:
                self.gnn_optimizer.zero_grad()
                
                output = self.gnn(
                    batch.x,
                    batch.edge_index,
                    batch.edge_attr,
                    batch.batch
                )
                
                loss = F.mse_loss(output, batch.y)
                loss.backward()
                self.gnn_optimizer.step()
                
                epoch_loss += loss.item()
                num_batches += 1
            
            avg_loss = epoch_loss / num_batches
            history.append({'epoch': epoch, 'loss': avg_loss})
            
            # Early stopping
            if avg_loss < best_loss:
                best_loss = avg_loss
                patience_counter = 0
            else:
                patience_counter += 1
            
            if patience_counter >= early_stop_patience:
                print(f"Early stopping at epoch {epoch}")
                break
            
            if epoch % 10 == 0:
                print(f"Epoch {epoch}: Loss = {avg_loss:.6f}")
        
        return history
    
    def train_pinn(self, X: torch.Tensor, y: torch.Tensor,
                   num_epochs: int = None,
                   physics_weight: float = None) -> List[Dict]:
        """Entrena el componente PINN"""
        num_epochs = num_epochs or CONFIG.model.NUM_EPOCHS
        physics_weight = physics_weight or CONFIG.model.PINN_WEIGHT_PHYSICS
        
        self.pinn.train()
        history = []
        
        for epoch in range(num_epochs):
            self.pinn_optimizer.zero_grad()
            
            loss, metrics = self.pinn.total_loss(
                X, y, physics_weight
            )
            
            loss.backward()
            self.pinn_optimizer.step()
            
            history.append({
                'epoch': epoch,
                **metrics
            })
            
            if epoch % 20 == 0:
                print(f"Epoch {epoch}: MSE={metrics['mse']:.6f}, "
                      f"Physics={metrics['physics']:.6f}, "
                      f"Total={metrics['total']:.6f}")
        
        return history
    
    def predict_route_score(self, graph_data: Data) -> float:
        """Predice score de optimalidad de ruta"""
        self.gnn.eval()
        with torch.no_grad():
            score = self.gnn(
                graph_data.x,
                graph_data.edge_index,
                graph_data.edge_attr,
                graph_data.batch
            )
        return float(score.item())
    
    def predict_bog_rate(self, conditions: Dict) -> float:
        """
        Predice tasa de Boil-Off Gas
        
        Args:
            conditions: {
                'ambient_temp': float,
                'sea_temp': float,
                'wind_speed': float,
                'tank_pressure': float,
                'insulation_quality': float,
                'cargo_level': float,
                'voyage_duration': float,
                'latitude': float
            }
        """
        self.pinn.eval()
        
        # Convertir a tensor
        x = torch.tensor([
            conditions['ambient_temp'],
            conditions['sea_temp'],
            conditions['wind_speed'],
            conditions['tank_pressure'],
            conditions['insulation_quality'],
            conditions['cargo_level'],
            conditions['voyage_duration'],
            conditions['latitude']
        ], dtype=torch.float32).unsqueeze(0)
        
        with torch.no_grad():
            bog_rate = self.pinn(x)
        
        return float(bog_rate.item())
    
    def assess_risk(self, conditions: Dict) -> Dict:
        """Evalúa riesgo marítimo usando red bayesiana"""
        model = self.bayesian.create_maritime_risk_model()
        
        # Ajustar con datos sintéticos si no hay datos reales
        if self.bayesian.available and model:
            evidence = {
                'wind_speed': 'high' if conditions.get('wind_speed', 0) > 20 else 'low',
                'visibility': 'good' if conditions.get('visibility', 10) > 5 else 'poor',
                'crew_experience': 'high',
                'maintenance_status': 'good'
            }
            return self.bayesian.query_risk(model, evidence)
        
        return {'low_risk': 0.7, 'medium_risk': 0.2, 'high_risk': 0.1}

"""
MARITIME-AI Training Pipeline
Pipeline completo de entrenamiento y evaluación
"""

import torch

from typing import Dict, List, Tuple

from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt
from config import CONFIG
from models.hybrid_architecture import MaritimeHybridModel


class MaritimeTrainingPipeline:
    """Pipeline de entrenamiento para Maritime-AI"""
    
    def __init__(self, model: MaritimeHybridModel = None):
        self.model = model or MaritimeHybridModel()
        self.scaler = StandardScaler()
        self.results_dir = Path('./results/')
        self.models_dir = Path('./models/saved/')
        
        # Crear directorios
        self.results_dir.mkdir(parents=True, exist_ok=True)
        self.models_dir.mkdir(parents=True, exist_ok=True)
    
    def prepare_graph_data(self, routes_ pd.DataFrame) -> List[Data]:
        """
        Prepara datos de rutas en formato PyTorch Geometric
        
        Args:
            routes_data: DataFrame con información de rutas
        """
        graphs = []
        
        # Agrupar por ruta/viaje
        for route_id in routes_data['route_id'].unique():
            route_df = routes_data[routes_data['route_id'] == route_id]
            
            # Features de nodos (puertos/waypoints)
            node_features = torch.tensor(
                route_df[['lat', 'lon', 'depth', 'current_speed', 
                         'current_direction', 'traffic_density',
                         'port_facilities', 'fuel_availability',
                         'weather_exposure', 'distance_to_shore']].values,
                dtype=torch.float32
            )
            
            # Crear grafo completo (todos conectados)
            num_nodes = len(route_df)
            edge_index = torch.combinations(
                torch.arange(num_nodes), r=2
            ).t().contiguous()
            
            # Features de aristas (distancia, corriente, etc.)
            edge_features = []
            for edge in edge_index.t():
                i, j = edge.tolist()
                dist = np.sqrt(
                    (route_df.iloc[i]['lat'] - route_df.iloc[j]['lat'])**2 +
                    (route_df.iloc[i]['lon'] - route_df.iloc[j]['lon'])**2
                )
                edge_features.append([
                    dist,
                    abs(route_df.iloc[i]['current_speed'] - route_df.iloc[j]['current_speed']),
                    route_df.iloc[i]['fuel_cost'],
                    route_df.iloc[i]['time_estimate'],
                    route_df.iloc[i]['risk_score']
                ])
            
            edge_features = torch.tensor(edge_features, dtype=torch.float32)
            
            # Target (score de optimalidad)
            y = torch.tensor([[route_df['optimal_score'].mean()]], dtype=torch.float32)
            
            # Crear objeto Data
            graph = Data(
                x=node_features,
                edge_index=edge_index,
                edge_attr=edge_features,
                y=y
            )
            
            graphs.append(graph)
        
        return graphs
    
    def prepare_pinn_data(self, bog_data: pd.DataFrame) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Prepara datos para entrenamiento PINN
        
        Args:
            bog_ DataFrame con datos de Boil-Off Gas
        """
        # Features de entrada
        X = bog_data[[
            'ambient_temperature',
            'sea_temperature',
            'wind_speed',
            'tank_pressure_bar'
        ]].values
        
        # Añadir features derivadas
        X_extended = np.column_stack([
            X,
            np.ones(len(X)) * 0.8,  # insulation_quality (simulado)
            np.ones(len(X)) * 0.7,  # cargo_level (simulado)
            np.arange(len(X)),      # voyage_duration
            np.ones(len(X)) * 30    # latitude (simulado)
        ])
        
        # Target
        y = bog_data['bog_rate'].values.reshape(-1, 1)
        
        # Escalar
        X_scaled = self.scaler.fit_transform(X_extended)
        
        return (
            torch.tensor(X_scaled, dtype=torch.float32),
            torch.tensor(y, dtype=torch.float32)
        )
    
    def train_full_pipeline(self, routes_data: pd.DataFrame,
                           bog_ pd.DataFrame,
                           validation_split: float = 0.2) -> Dict:
        """
        Ejecuta entrenamiento completo del modelo híbrido
        
        Args:
            routes_ Datos de rutas para GNN
            bog_ Datos BOG para PINN
            validation_split: Fracción para validación
        """
        print("=" * 60)
        print("MARITIME-AI TRAINING PIPELINE")
        print("=" * 60)
        
        results = {
            'timestamp': datetime.now().isoformat(),
            'gnn_history': [],
            'pinn_history': [],
            'metrics': {}
        }
        
        # 1. Preparar datos GNN
        print("\n[1/4] Preparing GNN graph data...")
        graphs = self.prepare_graph_data(routes_data)
        
        # Split train/val
        train_graphs = graphs[:int(len(graphs) * (1 - validation_split))]
        val_graphs = graphs[int(len(graphs) * (1 - validation_split)):]
        
        # DataLoader
        train_loader = DataLoader(train_graphs, batch_size=CONFIG.model.BATCH_SIZE, shuffle=True)
        
        # 2. Entrenar GNN
        print("\n[2/4] Training GNN component...")
        self.model.compile()
        gnn_history = self.model.train_gnn(train_loader)
        results['gnn_history'] = gnn_history
        
        # 3. Preparar datos PINN
        print("\n[3/4] Preparing PINN training data...")
        X_pinn, y_pinn = self.prepare_pinn_data(bog_data)
        
        # Split
        X_train, X_val, y_train, y_val = train_test_split(
            X_pinn, y_pinn, test_size=validation_split, random_state=42
        )
        
        # 4. Entrenar PINN
        print("\n[4/4] Training PINN component...")
        pinn_history = self.model.train_pinn(X_train, y_train)
        results['pinn_history'] = pinn_history
        
        # 5. Evaluar
        print("\n[Evaluation] Computing final metrics...")
        
        # GNN evaluation
        self.model.gnn.eval()
        gnn_test_loss = 0
        with torch.no_grad():
            for graph in val_graphs:
                pred = self.model.gnn(graph.x, graph.edge_index, graph.edge_attr, graph.batch)
                gnn_test_loss += F.mse_loss(pred, graph.y).item()
        
        # PINN evaluation
        self.model.pinn.eval()
        with torch.no_grad():
            pinn_pred = self.model.pinn(X_val)
            pinn_mse = F.mse_loss(pinn_pred, y_val).item()
        
        results['metrics'] = {
            'gnn_val_loss': gnn_test_loss / len(val_graphs) if val_graphs else 0,
            'pinn_val_mse': pinn_mse,
            'gnn_train_epochs': len(gnn_history),
            'pinn_train_epochs': len(pinn_history)
        }
        
        # 6. Guardar resultados
        self._save_results(results)
        self._save_models()
        self._plot_training_history(results)
        
        print("\n" + "=" * 60)
        print("TRAINING COMPLETE")
        print("=" * 60)
        print(f"GNN Validation Loss: {results['metrics']['gnn_val_loss']:.6f}")
        print(f"PINN Validation MSE: {results['metrics']['pinn_val_mse']:.6f}")
        print(f"Results saved to: {self.results_dir}")
        print(f"Models saved to: {self.models_dir}")
        
        return results
    
    def _save_results(self, results: Dict):
        """Guarda resultados en JSON"""
        # Convertir tensors a listas para JSON
        results_serializable = json.loads(json.dumps(results, default=lambda x: x.item() if hasattr(x, 'item') else str(x)))
        
        with open(self.results_dir / 'training_results.json', 'w') as f:
            json.dump(results_serializable, f, indent=2)
    
    def _save_models(self):
        """Guarda modelos entrenados"""
        torch.save(self.model.gnn.state_dict(), self.models_dir / 'gnn_model.pt')
        torch.save(self.model.pinn.state_dict(), self.models_dir / 'pinn_model.pt')
        torch.save(self.scaler, self.models_dir / 'scaler.pkl')
    
    def _plot_training_history(self, results: Dict):
        """Genera gráficos de historial de entrenamiento"""
        fig, axes = plt.subplots(1, 2, figsize=(15, 5))
        
        # GNN Loss
        if results['gnn_history']:
            epochs = [h['epoch'] for h in results['gnn_history']]
            losses = [h['loss'] for h in results['gnn_history']]
            axes[0].plot(epochs, losses, 'b-', linewidth=2)
            axes[0].set_xlabel('Epoch')
            axes[0].set_ylabel('Loss')
            axes[0].set_title('GNN Training Loss')
            axes[0].grid(True, alpha=0.3)
        
        # PINN Loss
        if results['pinn_history']:
            epochs = [h['epoch'] for h in results['pinn_history']]
            mse_losses = [h['mse'] for h in results['pinn_history']]
            physics_losses = [h['physics'] for h in results['pinn_history']]
            total_losses = [h['total'] for h in results['pinn_history']]
            
            axes[1].plot(epochs, mse_losses, 'b-', label='MSE', linewidth=2)
            axes[1].plot(epochs, physics_losses, 'r--', label='Physics', linewidth=2)
            axes[1].plot(epochs, total_losses, 'g-', label='Total', linewidth=2)
            axes[1].set_xlabel('Epoch')
            axes[1].set_ylabel('Loss')
            axes[1].set_title('PINN Training Loss')
            axes[1].legend()
            axes[1].grid(True, alpha=0.3)
        
        plt.tight_layout()
        plt.savefig(self.results_dir / 'training_history.png', dpi=300)
        plt.close()

export const apiUtils = {
  // Formatear coordenadas
  formatCoordinates: (lat, lon) => `${lat.toFixed(4)}, ${lon.toFixed(4)}`,
  
  // Calcular distancia haversine
  calculateDistance: (lat1, lon1, lat2, lon2) => {
    const R = 6371 // Radio terrestre km
    const dLat = ((lat2 - lat1) * Math.PI) / 180
    const dLon = ((lon2 - lon1) * Math.PI) / 180
    const a =
      Math.sin(dLat / 2) * Math.sin(dLat / 2) +
      Math.cos((lat1 * Math.PI) / 180) *
        Math.cos((lat2 * Math.PI) / 180) *
        Math.sin(dLon / 2) *
        Math.sin(dLon / 2)
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
    return R * c
  },

  // Formatear fecha para API
  formatDate: (date) => date.toISOString().split('T')[0],

/**
 * Custom Hook para gestión de datos marítimos
 */

import { useState, useEffect, useCallback } from 'react'
import { maritimeAI } from '../utils/api'

export function useMaritimeData(dataType, params) {
  const [data, setData] = useState(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  const fetchData = useCallback(async () => {
    setLoading(true)
    setError(null)

    try {
      let response

      switch (dataType) {
        case 'ocean':
          response = await maritimeAI.getOceanData(params.bbox, params.timeRange)
          break
        case 'ais':
          response = await maritimeAI.getAISTrajectory(
            params.mmsi,
            params.startDate,
            params.endDate
          )
          break
        case 'lng':
          response = await maritimeAI.getLNGTerminalData(params.terminalId)
          break
        default:
          throw new Error(`Unknown data type: ${dataType}`)
      }

      setData(response)
    } catch (err) {
      setError(err.message || 'Error fetching data')
    } finally {
      setLoading(false)
    }
  }, [dataType, params])

  useEffect(() => {
    fetchData()
  }, [fetchData])

  const refresh = useCallback(() => {
    fetchData()
  }, [fetchData])

  return { data, loading, error, refresh }
}

export function useModelInference() {
  const [loading, setLoading] = useState(false)
  const [result, setResult] = useState(null)
  const [error, setError] = useState(null)

  const predictBOG = useCallback(async (conditions) => {
    setLoading(true)
    setError(null)

    try {
      const response = await maritimeAI.predictBOG(conditions)
      setResult(response)
      return response
    } catch (err) {
      setError(err.message || 'Error predicting BOG')
      throw err
    } finally {
      setLoading(false)
    }
  }, [])

  const optimizeRoute = useCallback(async (routeData) => {
    setLoading(true)
    setError(null)

    try {
      const response = await maritimeAI.optimizeRoute(routeData)
      setResult(response)
      return response
    } catch (err) {
      setError(err.message || 'Error optimizing route')
      throw err
    } finally {
      setLoading(false)
    }
  }, [])

  const assessRisk = useCallback(async (evidence) => {
    setLoading(true)
    setError(null)

    try {
      const response = await maritimeAI.assessRisk(evidence)
      setResult(response)
      return response
    } catch (err) {
      setError(err.message || 'Error assessing risk')
      throw err
    } finally {
      setLoading(false)
    }
  }, [])

  return {
    predictBOG,
    optimizeRoute,
    assessRisk,
    loading,
    result,
    error,
  }
}

export default { useMaritimeData, useModelInference }

#### **Problema a Resolver:**
La industria del Gas Natural Licuado (LNG) enfrenta dos desafíos críticos:
1. **Pérdidas por Boil-Off Gas (BOG):** Durante el transporte marítimo, el LNG se evapora naturalmente (0.1-0.3%/día), representando pérdidas económicas de millones de dólares anuales y emisiones de metano significativas .
2. **Ineficiencia en Rutas Marítimas:** Las rutas subóptimas incrementan el consumo de combustible, tiempo de tránsito y exposición a condiciones oceanográficas adversas .

#### **Justificación del Impacto:**
- **Económico:** Reducción del 15-25% en pérdidas por BOG mediante predicción precisa .
- **Ambiental:** Disminución de emisiones de CO₂ y metano, alineado con regulaciones IMO 2025 .
- **Operacional:** Optimización de rutas en tiempo real considerando corrientes, oleaje y condiciones meteorológicas .
- **Seguridad:** Evaluación de riesgos marítimos mediante Redes Bayesianas para prevención de accidentes .

            end_date: Fecha fin (YYYY-MM-DD)

                        'air_temperature': float(parts[13]) if parts[13] != 'MM' else np.nan,

                    })

class AISDataIngestor:
    """Ingestor de datos AIS (Automatic Identification System)"""
    
    def __init__(self, data_path: str = None):
        self.data_path = data_path or CONFIG.data.AIS_DATA_PATH
        
    def load_ais_trajectory(self, mmsi: int, 
                           start_date: str, 
                           end_date: str) -> pd.DataFrame:
        """
        Carga trayectorias AIS para un buque específico
        
        Args:
            mmsi: Maritime Mobile Service Identity del buque
            start_date: Fecha inicio
            end_date: Fecha fin
        """
        # Buscar archivos AIS
        ais_files = list(Path(self.data_path).glob(f"*{mmsi}*.csv"))
        
        if not ais_files:
            print(f"No AIS files found for MMSI {mmsi}")
            return pd.DataFrame()
        
        dfs = []
        for file in ais_files:
            try:
                df = pd.read_csv(file)
                dfs.append(df)
            except Exception as e:
                print(f"Error reading {file}: {e}")
        
        if dfs:
            combined = pd.concat(dfs, ignore_index=True)
            combined['timestamp'] = pd.to_datetime(combined['timestamp'])
            mask = (combined['timestamp'] >= start_date) & \
                   (combined['timestamp'] <= end_date)
            return combined[mask]
        
        return pd.DataFrame()
    
    def extract_vessel_features(self, ais_df: pd.DataFrame) -> Dict:
        """Extrae características del buque desde datos AIS"""
        if ais_df.empty:
            return {}
        
        return {
            'mmsi': ais_df['mmsi'].iloc[0] if 'mmsi' in ais_df.columns else None,
            'total_distance_km': self._calculate_distance(ais_df),
            'avg_speed_knots': ais_df['sog'].mean() if 'sog' in ais_df.columns else None,
            'max_speed_knots': ais_df['sog'].max() if 'sog' in ais_df.columns else None,
            'course_changes': self._count_course_changes(ais_df),
            'ports_visited': len(ais_df['destination'].unique()) if 'destination' in ais_df.columns else 0
        }
    
    def _calculate_distance(self, df: pd.DataFrame) -> float:
        """Calcula distancia total recorrida (km)"""
        if len(df) < 2:
            return 0.0
        
        from math import radians, sin, cos, sqrt, atan2
        
        def haversine(lat1, lon1, lat2, lon2):
            R = 6371  # Radio terrestre km
            dlat = radians(lat2 - lat1)
            dlon = radians(lon2 - lon1)
            a = sin(dlat/2)**2 + cos(radians(lat1)) * cos(radians(lat2)) * sin(dlon/2)**2
            c = 2 * atan2(sqrt(a), sqrt(1-a))
            return R * c
        
        total = 0
        for i in range(1, len(df)):
            if 'latitude' in df.columns and 'longitude' in df.columns:
                total += haversine(
                    df.iloc[i-1]['latitude'], df.iloc[i-1]['longitude'],
                    df.iloc[i]['latitude'], df.iloc[i]['longitude']
                )
        return total
    
    def _count_course_changes(self, df: pd.DataFrame, threshold: float = 10) -> int:
        """Cuenta cambios significativos de rumbo"""
        if 'cog' not in df.columns or len(df) < 2:
            return 0
        
        changes = 0
        for i in range(1, len(df)):
            diff = abs(df.iloc[i]['cog'] - df.iloc[i-1]['cog'])
            if diff > 180:
                diff = 360 - diff
            if diff > threshold:
                changes += 1
        return changes

        Genera datos sintéticos de BOG para testing
        
        Basado en modelos físicos de transferencia de calor 
        """
        dates = pd.date_range(start='2025-01-01', periods=num_days, freq='D')
        
        # Factores que afectan BOG
        ambient_temp = 15 + 10 * np.sin(np.linspace(0, 2*np.pi, num_days))  # Variación estacional
        sea_temp = 12 + 5 * np.sin(np.linspace(0, 2*np.pi, num_days))
        wind_speed = np.random.uniform(5, 25, num_days)
        
        # Modelo simplificado de BOG
        # BOG rate = base_rate + f(ambient_temp, sea_temp, wind, insulation)
        base_rate = CONFIG.lng.BOG_BASELINE_RATE
        temp_factor = 0.0001 * (ambient_temp - sea_temp)
        wind_factor = 0.00005 * wind_speed
        noise = np.random.normal(0, 0.0002, num_days)
        
        bog_rate = base_rate + temp_factor + wind_factor + noise
        bog_rate = np.clip(bog_rate, 0.0005, CONFIG.lng.BOG_MAX_ACCEPTABLE)
        
        bog_volume = bog_rate * tank_volume  # m³/día
        
        df = pd.DataFrame({
            'timestamp': dates,
            'ambient_temperature': ambient_temp,
            'sea_temperature': sea_temp,
            'wind_speed': wind_speed,
            'bog_rate': bog_rate,
            'bog_volume_m3': bog_volume,
            'tank_pressure_bar': 0.15 + 0.05 * np.random.uniform(0, 1, num_days)
        })
        
        return df

                print(f"Saved {name} with {len(df)} records")
        
        return results

"""
MARITIME-AI Hybrid Model Architecture
Arquitectura híbrida: GNN + PINN + Bayesian Networks
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
from typing import Dict, List, Tuple, Optional
from torch_geometric.nn import GATConv, GINConv, global_mean_pool
from torch_geometric.data import Data, DataLoader
import xarray as xr
from config import CONFIG


class MaritimeGNN(nn.Module):
    """
    Graph Neural Network para optimización de rutas marítimas
    
    Usa Graph Attention Networks (GAT) para modelar:
    - Nodos: Puertos, waypoints, condiciones oceanográficas
    - Aristas: Rutas posibles, corrientes, restricciones
    """
    
    def __init__(self, node_features: int, edge_features: int, 
                 hidden_dim: int = None, num_layers: int = None,
                 num_heads: int = None, dropout: float = None):
        super().__init__()
        
        # Usar configuración global si no se proporciona
        hidden_dim = hidden_dim or CONFIG.model.GNN_HIDDEN_DIM
        num_layers = num_layers or CONFIG.model.GNN_NUM_LAYERS
        num_heads = num_heads or CONFIG.model.GNN_HEADS
        dropout = dropout or CONFIG.model.GNN_DROPOUT
        
        self.node_encoder = nn.Linear(node_features, hidden_dim)
        self.edge_encoder = nn.Linear(edge_features, hidden_dim)
        
        # Capas GAT (Graph Attention)
        self.gat_layers = nn.ModuleList()
        for i in range(num_layers):
            in_dim = hidden_dim if i == 0 else hidden_dim * num_heads
            self.gat_layers.append(
                GATConv(
                    in_channels=in_dim,
                    out_channels=hidden_dim,
                    heads=num_heads if i < num_layers - 1 else 1,
                    dropout=dropout,
                    concat=i < num_layers - 1
                )
            )
        
        # Capas fully connected para predicción
        self.fc_layers = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(hidden_dim // 2, hidden_dim // 4),
            nn.ReLU(),
            nn.Linear(hidden_dim // 4, 1)  # Score de ruta óptima
        )
        
    def forward(self, x: torch.Tensor, edge_index: torch.Tensor,
                edge_attr: torch.Tensor, batch: torch.Tensor) -> torch.Tensor:
        """
        Forward pass del GNN
        
        Args:
            x: Features de nodos [num_nodes, node_features]
            edge_index: Conectividad del grafo [2, num_edges]
            edge_attr: Features de aristas [num_edges, edge_features]
            batch: Batch assignment para graph pooling
        """
        # Encoding inicial
        x = self.node_encoder(x)
        edge_attr = self.edge_encoder(edge_attr)
        
        # Pasar por capas GAT
        for gat_layer in self.gat_layers:
            x = gat_layer(x, edge_index, edge_attr=edge_attr)
            x = F.relu(x)
        
        # Global pooling
        x = global_mean_pool(x, batch)
        
        # Predicción final
        output = self.fc_layers(x)
        return output


class PhysicsInformedNN(nn.Module):
    """
    Physics-Informed Neural Network para predicción de Boil-Off Gas
    
    Incorpora la ecuación de transferencia de calor como regularización:
    ∂T/∂t = α∇²T + Q/(ρ·cp)
    
    Donde:
    - T: Temperatura
    - α: Difusividad térmica
    - Q: Fuente de calor
    - ρ: Densidad
    - cp: Calor específico
    """
    
    def __init__(self, input_dim: int, hidden_layers: List[int] = None,
                 activation: str = None):
        super().__init__()
        
        hidden_layers = hidden_layers or CONFIG.model.PINN_HIDDEN_LAYERS
        activation = activation or CONFIG.model.PINN_ACTIVATION
        
        # Construir arquitectura
        layers = []
        prev_dim = input_dim
        
        for hidden_dim in hidden_layers:
            layers.append(nn.Linear(prev_dim, hidden_dim))
            if activation == "tanh":
                layers.append(nn.Tanh())
            elif activation == "relu":
                layers.append(nn.ReLU())
            elif activation == "swish":
                layers.append(nn.SiLU())
            prev_dim = hidden_dim
        
        layers.append(nn.Linear(prev_dim, 1))  # Output: BOG rate
        
        self.network = nn.Sequential(*layers)
        
        # Parámetros físicos (aprendibles)
        self.thermal_diffusivity = nn.Parameter(torch.tensor(1.0e-7))
        self.heat_transfer_coef = nn.Parameter(torch.tensor(10.0))
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """Forward pass estándar"""
        return self.network(x)
    
    def compute_physics_loss(self, x: torch.Tensor, 
                            predicted: torch.Tensor) -> torch.Tensor:
        """
        Calcula el término de pérdida basado en física
        
        Implementa la ecuación de calor como regularización
        """
        x.requires_grad_(True)
        predicted = self.network(x)
        
        # Gradientes automáticos
        grad_outputs = torch.ones_like(predicted)
        gradients = torch.autograd.grad(
            outputs=predicted,
            inputs=x,
            grad_outputs=grad_outputs,
            create_graph=True,
            retain_graph=True
        )[0]
        
        # Laplaciano (simplificado para 1D)
        laplacian = torch.zeros_like(predicted)
        for i in range(x.shape[1]):
            grad_i = gradients[:, i:i+1]
            grad_i.requires_grad_(True)
            laplacian_i = torch.autograd.grad(
                outputs=grad_i,
                inputs=x,
                grad_outputs=torch.ones_like(grad_i),
                create_graph=True
            )[0][:, i:i+1]
            laplacian += laplacian_i
        
        # Ecuación de calor residual
        # ∂T/∂t - α∇²T = 0 (en estado estacionario)
        physics_residual = -self.thermal_diffusivity * laplacian
        
        return torch.mean(physics_residual ** 2)
    
    def total_loss(self, x: torch.Tensor, y_true: torch.Tensor,
                   physics_weight: float = None) -> Tuple[torch.Tensor, Dict]:
        """
        Pérdida total: MSE + Physics regularization
        
        Args:
            x: Inputs (condiciones ambientales, propiedades tanque)
            y_true: BOG rate observado
            physics_weight: Peso del término físico
        """
        physics_weight = physics_weight or CONFIG.model.PINN_WEIGHT_PHYSICS
        
        predicted = self.forward(x)
        
        # MSE loss
        mse_loss = F.mse_loss(predicted, y_true)
        
        # Physics loss
        physics_loss = self.compute_physics_loss(x, predicted)
        
        # Loss total
        total_loss = mse_loss + physics_weight * physics_loss
        
        metrics = {
            'mse': mse_loss.item(),
            'physics': physics_loss.item(),
            'total': total_loss.item()
        }
        
        return total_loss, metrics


class BayesianRiskNetwork:
    """
    Red Bayesiana para evaluación de riesgos marítimos
    
    Variables:
    - Condiciones meteorológicas (viento, oleaje, visibilidad)
    - Estado del buque (velocidad, carga, mantenimiento)
    - Factores humanos (experiencia tripulación, fatiga)
    - Riesgo de accidente (colisión, varada, incendio)
    """
    
    def __init__(self):
        try:
            from pgmpy.models import BayesianNetwork
            from pgmpy.estimators import HillClimbSearch, BicScore
            from pgmpy.estimators import MaximumLikelihoodEstimator, BayesianEstimator
            self.HillClimbSearch = HillClimbSearch
            self.BicScore = BicScore
            self.MLE = MaximumLikelihoodEstimator
            self.BayesEst = BayesianEstimator
            self.BayesianNetwork = BayesianNetwork
            self.available = True
        except ImportError:
            print("pgmpy not installed. Install with: pip install pgmpy")
            self.available = False
    
    def create_maritime_risk_model(self) -> 'BayesianNetwork':
        """Crea la estructura de la red bayesiana"""
        if not self.available:
            return None
        
        # Definir aristas (relaciones causales)
        edges = [
            ('wind_speed', 'wave_height'),
            ('wind_speed', 'vessel_stability'),
            ('wave_height', 'vessel_stability'),
            ('wave_height', 'navigation_difficulty'),
            ('visibility', 'navigation_difficulty'),
            ('vessel_stability', 'accident_risk'),
            ('navigation_difficulty', 'accident_risk'),
            ('crew_experience', 'accident_risk'),
            ('maintenance_status', 'vessel_stability'),
            ('maintenance_status', 'accident_risk'),
            ('cargo_load', 'vessel_stability'),
            ('sea_current', 'navigation_difficulty'),
            ('accident_risk', 'route_safety_score')
        ]
        
        model = self.BayesianNetwork(edges)
        return model
    
    def fit_from_data(self, data: pd.DataFrame) -> 'BayesianNetwork':
        """Ajusta la red bayesiana desde datos"""
        if not self.available:
            return None
        
        import pandas as pd
        
        model = self.create_maritime_risk_model()
        
        # Aprendizaje de estructura (opcional, si no se define manualmente)
        # hc = self.HillClimbSearch(data)
        # best_model = hc.estimate(scoring_method=self.BicScore(data))
        
        # Estimación de parámetros
        model.fit(data, estimator=self.BayesEst, prior_type="BDeu")
        
        return model
    
    def query_risk(self, model: 'BayesianNetwork', 
                   evidence: Dict) -> Dict:
        """
        Consulta la probabilidad de riesgo dado evidencia
        
        Args:
            model: Red bayesiana entrenada
            evidence: Dict con variables observadas
        """
        if not self.available or model is None:
            return {}
        
        from pgmpy.inference import VariableElimination
        
        inference = VariableElimination(model)
        
        # Consultar probabilidad de accidente
        result = inference.query(
            variables=['accident_risk'],
            evidence=evidence
        )
        
        return {
            'low_risk': float(result.values[0]),
            'medium_risk': float(result.values[1]),
            'high_risk': float(result.values[2])
        }


class MaritimeHybridModel:
    """
    Modelo híbrido principal que integra GNN + PINN + Bayesian
    
    Orquesta los tres componentes para:
    1. Optimizar rutas (GNN)
    2. Predecir BOG (PINN)
    3. Evaluar riesgos (Bayesian)
    """
    
    def __init__(self, node_features: int = 10, edge_features: int = 5,
                 pinn_input_dim: int = 8):
        self.gnn = MaritimeGNN(node_features, edge_features)
        self.pinn = PhysicsInformedNN(pinn_input_dim)
        self.bayesian = BayesianRiskNetwork()
        
        self.gnn_optimizer = None
        self.pinn_optimizer = None
        
    def compile(self, gn_lr: float = None, pinn_lr: float = None):
        """Configura optimizadores"""
        gn_lr = gn_lr or CONFIG.model.LEARNING_RATE
        pinn_lr = pinn_lr or CONFIG.model.LEARNING_RATE
        
        self.gnn_optimizer = torch.optim.Adam(
            self.gnn.parameters(), lr=gn_lr
        )
        self.pinn_optimizer = torch.optim.Adam(
            self.pinn.parameters(), lr=pinn_lr
        )
    
    def train_gnn(self, dataloader: DataLoader, num_epochs: int = None,
                  early_stop_patience: int = None) -> List[Dict]:
        """Entrena el componente GNN"""
        num_epochs = num_epochs or CONFIG.model.NUM_EPOCHS
        early_stop_patience = early_stop_patience or CONFIG.model.EARLY_STOPPING_PATIENCE
        
        self.gnn.train()
        history = []
        best_loss = float('inf')
        patience_counter = 0
        
        for epoch in range(num_epochs):
            epoch_loss = 0
            num_batches = 0
            
            for batch in dataloader:
                self.gnn_optimizer.zero_grad()
                
                output = self.gnn(
                    batch.x,
                    batch.edge_index,
                    batch.edge_attr,
                    batch.batch
                )
                
                loss = F.mse_loss(output, batch.y)
                loss.backward()
                self.gnn_optimizer.step()
                
                epoch_loss += loss.item()
                num_batches += 1
            
            avg_loss = epoch_loss / num_batches
            history.append({'epoch': epoch, 'loss': avg_loss})
            
            # Early stopping
            if avg_loss < best_loss:
                best_loss = avg_loss
                patience_counter = 0
            else:
                patience_counter += 1
            
            if patience_counter >= early_stop_patience:
                print(f"Early stopping at epoch {epoch}")
                break
            
            if epoch % 10 == 0:
                print(f"Epoch {epoch}: Loss = {avg_loss:.6f}")
        
        return history
    
    def train_pinn(self, X: torch.Tensor, y: torch.Tensor,
                   num_epochs: int = None,
                   physics_weight: float = None) -> List[Dict]:
        """Entrena el componente PINN"""
        num_epochs = num_epochs or CONFIG.model.NUM_EPOCHS
        physics_weight = physics_weight or CONFIG.model.PINN_WEIGHT_PHYSICS
        
        self.pinn.train()
        history = []
        
        for epoch in range(num_epochs):
            self.pinn_optimizer.zero_grad()
            
            loss, metrics = self.pinn.total_loss(
                X, y, physics_weight
            )
            
            loss.backward()
            self.pinn_optimizer.step()
            
            history.append({
                'epoch': epoch,
                **metrics
            })
            
            if epoch % 20 == 0:
                print(f"Epoch {epoch}: MSE={metrics['mse']:.6f}, "
                      f"Physics={metrics['physics']:.6f}, "
                      f"Total={metrics['total']:.6f}")
        
        return history
    
    def predict_route_score(self, graph_ Data) -> float:
        """Predice score de optimalidad de ruta"""
        self.gnn.eval()
        with torch.no_grad():
            score = self.gnn(
                graph_data.x,
                graph_data.edge_index,
                graph_data.edge_attr,
                graph_data.batch
            )
        return float(score.item())
    
    def predict_bog_rate(self, conditions: Dict) -> float:
        """
        Predice tasa de Boil-Off Gas
        
        Args:
            conditions: {
                'ambient_temp': float,
                'sea_temp': float,
                'wind_speed': float,
                'tank_pressure': float,
                'insulation_quality': float,
                'cargo_level': float,
                'voyage_duration': float,
                'latitude': float
            }
        """
        self.pinn.eval()
        
        # Convertir a tensor
        x = torch.tensor([
            conditions['ambient_temp'],
            conditions['sea_temp'],
            conditions['wind_speed'],
            conditions['tank_pressure'],
            conditions['insulation_quality'],
            conditions['cargo_level'],
            conditions['voyage_duration'],
            conditions['latitude']
        ], dtype=torch.float32).unsqueeze(0)
        
        with torch.no_grad():
            bog_rate = self.pinn(x)
        
        return float(bog_rate.item())
    
    def assess_risk(self, conditions: Dict) -> Dict:
        """Evalúa riesgo marítimo usando red bayesiana"""
        model = self.bayesian.create_maritime_risk_model()
        
        # Ajustar con datos sintéticos si no hay datos reales
        if self.bayesian.available and model:
            evidence = {
                'wind_speed': 'high' if conditions.get('wind_speed', 0) > 20 else 'low',
                'visibility': 'good' if conditions.get('visibility', 10) > 5 else 'poor',
                'crew_experience': 'high',
                'maintenance_status': 'good'
            }
            return self.bayesian.query_risk(model, evidence)
        
        return {'low_risk': 0.7, 'medium_risk': 0.2, 'high_risk': 0.1}

    def prepare_graph_data(self, routes_ pd.DataFrame) -> List[Data]:
        """

        Args:
            routes_ DataFrame con información de rutas
        """

                    (route_df.iloc[i]['lat'] - route_df.iloc[j]['lat'])**2 +

                x=node_features,
